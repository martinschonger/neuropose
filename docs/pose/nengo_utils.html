 72def mex_hat(
 73    x: npt.ArrayLike,
 74    var_exc: np.float64,
 75    var_inh: np.float64,
 76    fact_exc: np.float64,
 77    fact_inh: np.float64,
 78    offset: np.float64
 79) -> npt.NDArray[np.float64]:
 80    """Computes a mexican hat function at the specified locations.
 81
 82    Args:
 83        x: Points for evaluating the Gaussians.
 84        var_exc: Variance of the excitatory Gaussian.
 85        var_inh: Variance of the inhibitory Gaussian.
 86        fact_exc: Scaling factor of the excitatory Gaussian.
 87        fact_inh: Scaling factor of the inhibitory Gaussian.
 88        offset: Constant global offset.
 89
 90    Returns:
 91        Function values.
 92    """
 93    cov_exc = var_exc * np.eye(3)
 94    Z = multivariate_normal.pdf(x=x, mean=np.zeros(3), cov=cov_exc)
 95    Z = Z * fact_exc
 96
 97    cov_inh = var_inh * np.eye(3)
 98    Z2 = multivariate_normal.pdf(x=x, mean=np.zeros(3), cov=cov_inh)
 99    Z2 = Z2 * fact_inh
100
101    Z_final = Z - Z2 - offset
102
103    return Z_final.ravel()

144def rotate_to_grid(
145    coefs: npt.NDArray[np.float64],
146    grid_shape: GridShape,
147    fgrid_shape: GridShape
148) -> npt.NDArray[np.float64]:
149    """Propagates a base Fourier coefficient vector to all grid points via rotation in reciprocal
150    space.
151
152    Args:
153        coefs: Base Fourier coefficients for position ``(0,0,0)``.
154        grid_shape: Shape of a unit cell of the neuron lattice.
155        fgrid_shape: Shape of the discrete sampling grid for the FFT, and of the reciprocal grid.
156
157    Returns:
158        A four-dimensional array where the first three dimensions index the grid points, and the
159        last dimension corresponds to the Fourier coefficients.
160
161    Notes:
162        By exploiting the grid structure and NumPy's vectorization this function is very efficient.
163    """
164    res_u, res_v, res_w = grid_shape
165
166    center_u = get_domain() * np.array([1, 0, 0]) / np.array(grid_shape)
167    rot_mat_u = get_rotation_matrix(fgrid_shape, center_u)
168    rot_mat_u = csr_array(rot_mat_u)
169    center_v = get_domain() * np.array([0, 1, 0]) / np.array(grid_shape)
170    rot_mat_v = get_rotation_matrix(fgrid_shape, center_v)
171    rot_mat_v = csr_array(rot_mat_v)
172    center_w = get_domain() * np.array([0, 0, 1]) / np.array(grid_shape)
173    rot_mat_w = get_rotation_matrix(fgrid_shape, center_w)
174    rot_mat_w = csr_array(rot_mat_w)
175
176    res = np.zeros((*grid_shape, rot_mat_u.shape[0]))
177    res[0, 0, 0, :] = coefs.copy()
178
179    for mw in range(1, res_w):
180        res[0, 0, mw, :] = rot_mat_w.dot(res[0, 0, mw-1, :])
181
182    for mv in range(1, res_v):
183        res[0, mv, :, :] = rot_mat_v.dot(res[0, mv-1, :, :].transpose()).transpose()
184
185    for mu in range(1, res_u):
186        res[mu, :, :, :] = rot_mat_u.dot(res[mu-1, :, :, :].reshape((res_v*res_w, -1)
187                                                                    ).transpose()).transpose().reshape((res_v, res_w, -1))
188
189    return res

192def get_encoders(
193    grid_shape: GridShape,
194    fgrid_shape: GridShape,
195    cov: npt.NDArray[np.float64],
196    normalize: bool = True
197) -> npt.NDArray[np.float64]:
198    """Computes encoders as modified Fourier coefficient vectors of three-dimensional Gaussians.
199
200    Args:
201        grid_shape: Shape of a unit cell of the neuron lattice.
202        fgrid_shape: Shape of the discrete sampling grid for the FFT, and of the reciprocal grid.
203        cov: Covariance matrix of the Gaussian.
204        normalize: Whether to normalize the final encoder vectors to unit length. Defaults to True.
205
206    Returns:
207        Array with the encoder vectors as rows.
208    """
209    logger.info('create encoders_pose' + get_mem_info())
210    gauss0_f_cropped_flat = get_0_coefs(fgrid_shape, cov)
211
212    encoders = rotate_to_grid(gauss0_f_cropped_flat, grid_shape, fgrid_shape).reshape((np.prod(grid_shape), -1))
213
214    final_encoders = encoders[:, encoding_mask(fgrid_shape).ravel()]
215
216    scale_fact = 1
217    if normalize:
218        scale_fact = 1.0 / np.amax(np.linalg.norm(final_encoders, axis=1, keepdims=True))
219        final_encoders *= scale_fact
220
221    return final_encoders, scale_fact

253def shift_weights_vect_optimized(
254    base_coefs_on_grid: npt.NDArray[np.float64],
255    shift_vec: npt.NDArray[np.float64],
256    grid_shape: GridShape,
257    fgrid_shape: GridShape
258) -> npt.NDArray[np.float64]:
259    """Applies a constant shift vector to a set of Fourier coefficients.
260
261    The computation is performed in reciprocal space.
262
263    Args:
264        base_coefs_on_grid: Un-shifted Fourier coefficient vectors at each grid point.
265        shift_vec: Shift vector.
266        grid_shape: Shape of a unit cell of the neuron lattice.
267        fgrid_shape: Shape of the discrete sampling grid for the FFT, and of the reciprocal grid.
268
269    Returns:
270        Shifted weight matrix.
271    """
272    fspace_shifts = inv_fspace_base.dot(shift_vec)
273    center = get_domain() * fspace_shifts
274    rot_mat = get_rotation_matrix(fgrid_shape, center)
275    rot_mat = csr_array(rot_mat)
276
277    res = rot_mat.dot(base_coefs_on_grid.reshape((np.prod(grid_shape), -1)
278                                                 ).transpose()).transpose().reshape((*grid_shape, -1))
279
280    return res

283def shift_weights_angle_optimized(
284    base_coefs_on_grid: npt.NDArray[np.float64],
285    shift: np.float64,
286    grid_shape: GridShape,
287    fgrid_shape: GridShape
288) -> npt.NDArray[np.float64]:
289    """Shifts a set of Fourier coefficients based on the preferred orientation angle as specified by
290    the associated z-coordinates. Consequently, the base coefficients are shifted differently in
291    each z-plane.
292
293    The computation is performed in reciprocal space.
294
295    Args:
296        base_coefs_on_grid: Un-shifted Fourier coefficient vectors at each grid point.
297        shift: Shift value indicating how far the weights should be shifted along the preferred
298        direction of the individual z-planes.
299        grid_shape: Shape of a unit cell of the neuron lattice.
300        fgrid_shape: Shape of the discrete sampling grid for the FFT, and of the reciprocal grid.
301
302    Returns:
303        Transformed weight matrix.
304    """
305    res = np.zeros(base_coefs_on_grid.shape)
306
307    for i in range(grid_shape[2]):
308        layer_orientation_angle = i * 2 * pi / grid_shape[2]
309        shift_vec = polar_to_cartesian(shift, layer_orientation_angle)
310        fspace_shifts = inv_fspace_base.dot(shift_vec)
311        center = get_domain() * fspace_shifts
312        rot_mat = get_rotation_matrix(fgrid_shape, center)
313        rot_mat = csr_array(rot_mat)
314
315        res[:, :, i, :] = rot_mat.dot(base_coefs_on_grid[:, :, i, :].reshape(
316            (grid_shape[0] * grid_shape[1], -1)).transpose()).transpose().reshape((grid_shape[0], grid_shape[1], -1))
317
318    return res

321def _get_ifft_helper(grid_shape: GridShape):
322    def ifft_helper(x):
323        C_rotated_tmp = unflatten_coefs(x).reshape(grid_shape)
324        Z_rotated = np.fft.ifftn(C_rotated_tmp, s=grid_shape, norm='forward') / np.prod(get_domain())
325        Z_rotated = np.real(Z_rotated)
326        return Z_rotated
327    return ifft_helper

330def get_shift_con_weights(
331    grid_shape: GridShape,
332    tran_shift: np.float64,
333    frec_con_weights: npt.NDArray[np.float64],
334    rec_con_weights: npt.NDArray[np.float64]
335) -> npt.NDArray[np.float64]:
336    """Computes the weight matrix for the shift connection.
337
338    Args:
339        grid_shape: Shape of a unit cell of the neuron lattice.
340        tran_shift: Shift value indicating how far the weights should be shifted along the preferred
341        direction of the individual z-planes.
342        frec_con_weights: Fourier coefficient vector of the non-transposed rec_con_weights.
343        rec_con_weights: Final transposed rec_con_weights to be subtracted from the weight matrices
344        to counteract the rec_con_weights when shifting.
345
346    Returns:
347        Weight matrix.
348    """
349    # frec_con_weights is of the non-transposed weights
350    # rec_con_weights are the transposed weights
351    logger.info('get shift_con_weights' + get_mem_info())
352    fshift_con_weights = shift_weights_angle_optimized(frec_con_weights, tran_shift, grid_shape, grid_shape)
353    shift_con_weights = np.apply_along_axis(_get_ifft_helper(grid_shape), -1, fshift_con_weights)
354    shift_con_weights = shift_con_weights.reshape((np.prod(grid_shape), np.prod(grid_shape))).transpose()
355    shift_con_weights = shift_con_weights - rec_con_weights
356
357    return shift_con_weights

360def get_rot_con_weights(
361    grid_shape: GridShape,
362    rot_shift: np.float64,
363    frec_con_weights: npt.NDArray[np.float64],
364    rec_con_weights: npt.NDArray[np.float64]
365) -> tuple[npt.NDArray[np.float64], npt.NDArray[np.float64]]:
366    """Computes the weight matrices for the CW+CCW rotation connections.
367
368    Args:
369        grid_shape: Shape of a unit cell of the neuron lattice.
370        rot_shift: Shift value indicating how far up/down along the z-axis the weights should be
371        shifted.
372        frec_con_weights: Fourier coefficient vector of the non-transposed rec_con_weights.
373        rec_con_weights: Final transposed rec_con_weights to be subtracted from the weight matrices
374        to counteract the rec_con_weights when rotating.
375
376    Returns:
377        CW and CCW rotation weight matrices.
378    """
379    # frec_con_weights is of the non-transposed weights
380    # rec_con_weights are the transposed weights
381    logger.info('get rot_con_weights' + get_mem_info())
382    shift = np.array([0, 0, rot_shift])
383    frot_con_weights_pos = shift_weights_vect_optimized(frec_con_weights, shift, grid_shape, grid_shape)
384    rot_con_weights_pos = np.apply_along_axis(_get_ifft_helper(grid_shape), -1, frot_con_weights_pos)
385    rot_con_weights_pos = rot_con_weights_pos.reshape(
386        (np.prod(grid_shape), np.prod(grid_shape))).transpose()
387    rot_con_weights_pos = rot_con_weights_pos - rec_con_weights
388
389    frot_con_weights_neg = shift_weights_vect_optimized(frec_con_weights, -shift, grid_shape, grid_shape)
390    rot_con_weights_neg = np.apply_along_axis(_get_ifft_helper(grid_shape), -1, frot_con_weights_neg)
391    rot_con_weights_neg = rot_con_weights_neg.reshape(
392        (np.prod(grid_shape), np.prod(grid_shape))).transpose()
393    rot_con_weights_neg = rot_con_weights_neg - rec_con_weights
394
395    return rot_con_weights_pos, rot_con_weights_neg

398def get_weights_optimized(
399    grid_shape: GridShape,
400    weight_config: dict[np.float64]
401) -> tuple[npt.NDArray[np.float64], npt.NDArray[np.float64], npt.NDArray[np.float64], npt.NDArray[np.float64]]:
402    """Computes weight matrices for the recurrent attractor connection, shift connection, and CW+CCW
403    rotation connections.
404
405    Args:
406        grid_shape: Shape of a unit cell of the neuron lattice.
407        weight_config: Weight shape parameters, consisting of variance and factor of the excitatory
408        and inhibitory Gaussian components, and an offset.
409
410    Returns:
411        Four weight matrices.
412    """
413    logger.info('get weights' + get_mem_info())
414
415    grid_shape = tuple(grid_shape)
416    var_exc = weight_config['var_exc']
417    var_inh = weight_config['var_inh']
418    fact_exc = weight_config['fact_exc']
419    fact_inh = weight_config['fact_inh']
420    offset = weight_config['offset']
421    tran_shift = weight_config['tran_shift']
422    rot_shift = weight_config['rot_shift']
423
424    u_diff_unwrapped, v_diff_unwrapped, w_diff_unwrapped = get_3d_distances_unwrapped(grid_shape)
425    pos_unwrapped = np.stack((u_diff_unwrapped[0], v_diff_unwrapped[0], w_diff_unwrapped[0]), axis=-1)
426
427    Z_computed = mex_hat(pos_unwrapped, var_exc, var_inh, fact_exc, fact_inh, offset).reshape(grid_shape)
428
429    C_computed = np.fft.fftn(Z_computed, s=grid_shape, norm='forward') * np.prod(get_domain())
430    C_computed = flatten_coefs(C_computed)
431
432    logger.info('get rec_con_weights' + get_mem_info())
433    frec_con_weights = rotate_to_grid(C_computed, grid_shape, grid_shape)
434    rec_con_weights = np.apply_along_axis(_get_ifft_helper(grid_shape), -1, frec_con_weights)
435    rec_con_weights = rec_con_weights.reshape((np.prod(grid_shape), np.prod(grid_shape))).transpose()
436
437    shift_con_weights = get_shift_con_weights(grid_shape, tran_shift, frec_con_weights, rec_con_weights)
438
439    rot_con_weights_pos, rot_con_weights_neg = get_rot_con_weights(
440        grid_shape, rot_shift, frec_con_weights, rec_con_weights)
441
442    return rec_con_weights, shift_con_weights, rot_con_weights_pos, rot_con_weights_neg

445def get_attractor_network(
446    ens_cfg: nengo.Config,
447    tau: np.float64,
448    rec_con_weights: npt.NDArray[np.float64] = None,
449    **kwargs
450):
451    """Creates a Nengo ensemble and recurrent connection to obtain attractor dynamics.
452
453    Args:
454        ens_cfg: Parameters for the ensemble, including encoders.
455        tau: Synaptic time constant.
456        rec_con_weights: Recurrent connection weights. May be sparse. If not specified, the weights
457        are computed by Nengo based on the encoders. Defaults to None.
458        **kwargs: Parameters for the wrapper network, e.g. label or seed.
459
460    Returns:
461        A Nengo network containing the recurrently connected attractor ensemble.
462
463    Notes:
464        Specifies the block_shape config parameter for splitting the ensemble onto Loihi cores.
465    """
466    n_neurons = ens_cfg[nengo.Ensemble].encoders.shape[0]
467    dimensions = ens_cfg[nengo.Ensemble].encoders.shape[1]
468
469    with nengo.Network(**kwargs) as net, ens_cfg:
470        nengo_loihi.add_params(net)
471        net.attractor_ens = nengo.Ensemble(n_neurons,
472                                           dimensions,
473                                           label='attractor_network')
474        # interpret the ensemble as a three-dimensional grid and split into 4*4*4 blocks
475        net.config[net.attractor_ens].block_shape = nengo_loihi.BlockShape((4, 4, 4), (12, 12, 12))
476
477        # constant global inhibition
478        # net.global_inhib_node = nengo.Node(getf_global_inhib(dt))
479        # ginh = np.ones((dim_pose,1))*-10
480        # ginh[0,0] = -1
481        # nengo.Connection(net.global_inhib_node, net.pose, transform=ginh, synapse=tau)
482        # nengo.Connection(net.global_inhib_node, net.pose.neurons, transform=[[-1.0]] * net.pose.n_neurons, synapse=None)
483
484        # recurrent connection of the attractor network
485        if rec_con_weights is None:
486            net.rec_con = nengo.Connection(
487                net.attractor_ens,
488                net.attractor_ens,
489                # function=lambda x: x/tau,
490                # function=lambda x: x * 0.5,  # leaky integrator
491                solver=nengo.solvers.LstsqL2(weights=True),
492                synapse=tau,
493                label='attractor_dynamics')
494        else:
495            net.rec_con = nengo.Connection(
496                net.attractor_ens.neurons,
497                net.attractor_ens.neurons,
498                # transform=rec_con_weights,
499                transform=nengo.Sparse(rec_con_weights.shape, init=rec_con_weights),
500                synapse=tau,
501                label='attractor_dynamics')
502
503    return net

524def get_pose_network(
525    grid_shape: GridShape,
526    fgrid_shape: GridShape,
527    cov: npt.NDArray[np.float64],
528    tau: np.float64,
529    dt: np.float64,
530    rec_con_weights: npt.NDArray[np.float64] = None,
531    shift_con_weights: npt.NDArray[np.float64] = None,
532    rot_con_weights_pos: npt.NDArray[np.float64] = None,
533    rot_con_weights_neg: npt.NDArray[np.float64] = None,
534    _config=None,
535    **kwargs
536):
537    """Creates a Nengo network capable of performing path integration.
538
539    Multiple pose (position+orientation) estimates can be sustained simultaneously. Persistent reset
540    input can inhibit any other existing bumps and establish a single new bump at a desired
541    position.
542
543    Args:
544        grid_shape: Shape of a unit cell of the neuron lattice.
545        fgrid_shape: Shape of the discrete sampling grid for the FFT, and of the reciprocal grid.
546        cov: Covariance matrix of the Gaussian bumps.
547        tau: Synaptic time constant.
548        dt: Simulation time step in seconds.
549        rec_con_weights: Recurrent connection weights for the attractor ensemble. May be sparse. If
550        not specified, the weights are computed by Nengo based on the encoders. Defaults to None.
551        shift_con_weights: Connection weights for translation. May be sparse. If not specified, the
552        translation module is omitted. Defaults to None.
553        rot_con_weights_pos: Connection weights for CCW rotation. May be sparse. If not specified,
554        the CCW rotation module is omitted. Defaults to None.
555        rot_con_weights_neg: Connection weights for CW rotation. May be sparse. If not specified,
556        the CW rotation module is omitted. Defaults to None.
557
558    Returns:
559        A Nengo network containing the desired components for an attractor network, potentially
560        capable of performing path integration (depending on the parameters).
561    """
562    timesteps_per_sec = round(1/dt)
563    with nengo.Network(**kwargs) as net:
564        nengo_loihi.add_params(net)
565        dim_pose = encoding_len(fgrid_shape)
566        n_neurons = np.prod(np.array(grid_shape))

 81    """Computes a mexican hat function at the specified locations.
 82
 83    Args:
 84        x: Points for evaluating the Gaussians.
 85        var_exc: Variance of the excitatory Gaussian.
 86        var_inh: Variance of the inhibitory Gaussian.
 87        fact_exc: Scaling factor of the excitatory Gaussian.

 89        offset: Constant global offset.

 94    cov_exc = var_exc * np.eye(3)
 95    Z = multivariate_normal.pdf(x=x, mean=np.zeros(3), cov=cov_exc)
 96    Z = Z * fact_exc
 97

100    Z2 = Z2 * fact_inh
101
102    Z_final = Z - Z2 - offset

160        last dimension corresponds to the Fourier coefficients.

164    """

166

Creates a Nengo network capable of performing path integration.

Multiple pose (position+orientation) estimates can be sustained simultaneously. Persistent reset input can inhibit any other existing bumps and establish a single new bump at a desired position.

Arguments:
  • grid_shape: Shape of a unit cell of the neuron lattice.
  • fgrid_shape: Shape of the discrete sampling grid for the FFT, and of the reciprocal grid.
  • cov: Covariance matrix of the Gaussian bumps.
  • tau: Synaptic time constant.
  • dt: Simulation time step in seconds.
  • rec_con_weights: Recurrent connection weights for the attractor ensemble. May be sparse. If
  • not specified, the weights are computed by Nengo based on the encoders. Defaults to None.
  • shift_con_weights: Connection weights for translation. May be sparse. If not specified, the
  • translation module is omitted. Defaults to None.
  • rot_con_weights_pos: Connection weights for CCW rotation. May be sparse. If not specified,
  • the CCW rotation module is omitted. Defaults to None.
  • rot_con_weights_neg: Connection weights for CW rotation. May be sparse. If not specified,
  • the CW rotation module is omitted. Defaults to None.
Returns:

A Nengo network containing the desired components for an attractor network, potentially capable of performing path integration (depending on the parameters).
